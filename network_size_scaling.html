import pandas as pd
from graspologic.utils import cartesian_product


def numpy_n_bytes(n_nodes, n_edges):
    return n_nodes * n_nodes * 8


pairs = cartesian_product(n_nodes_range, n_edges_range)

rows = []

    n_bytes = numpy_n_bytes(n, m)
